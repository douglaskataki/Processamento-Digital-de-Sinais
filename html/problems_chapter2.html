P2.10

y(n) = x(n)+a*x(n-k) 1) Demonstraçao fora do escopo do Matlab 2)

nx = 0:200;
x = cos(0.2*pi*nx)+0.5*cos(0.6*pi*nx);
a = 0.1;
k = 50;
% x(n-k)
[xk,nk] = sigshift(x,nx,k);
% y(n) = x(n)+a*x(n-k)
[y,ny] = sigadd(x,nx,a*xk,nk);
% y(-k)
[yf,nyf] = sigfold(y,ny);
% r_yy = y(l)**y(-l)
[ryy,nryy] = conv_m(y,ny,yf,nyf);

subplot(311)
stem(nx,x);
xlabel('n');
ylabel('x(n)');
subplot(312)
stem(ny,y);
xlabel('n');
ylabel('y(n)');
subplot(313);
stem(nryy,ryy);
xlabel('n');
ylabel('r_{yy}(l)');
